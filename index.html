function doGet() { try { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Overall Record'); const cellValue = sheet.getRange('F2').getValue(); const response = { success: true, data: cellValue }; return ContentService.createTextOutput(JSON.stringify(response)).setMimeType(ContentService.MimeType.JSON); } catch (err) { const errorResponse = { success: false, error: err.toString() }; return ContentService.createTextOutput(JSON.stringify(errorResponse)).setMimeType(ContentService.MimeType.JSON); } }
